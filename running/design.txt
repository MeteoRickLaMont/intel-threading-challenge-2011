Compare for equality.

int _mm_testz_si128(__m128i s1, __m128i s2)

Returns 1 if the bitwise AND operation on s1 and s2 results in all zeros,
else returns 0. That is,

_mm_testz_si128 := ( (s1 & s2) == 0 ? 1 : 0 )

This intrinsic checks if the ZF flag equals 1 as a result of the instruction
PTEST s1, s2. For example, it allows you to check if all set bits in s2
(mask) are zeros in s1.

Corresponding instruction: PTEST

int _mm_testc_si128 (__m128i s1, __m128i s2)

Returns 1 if the bitwise AND operation on s2 and logical NOT s1 results
in all zeros, else returns 0. That is,

_mm_testc_si128 := ( (~s1 & s2) == 0 ? 1 : 0 )

This intrinsic checks if the CF flag equals 1 as a result of the instruction
PTEST s1, s2. For example it allows you to check if all set bits in s2
(mask) are also set in s1.

Corresponding instruction: PTEST

int _mm_testnzc_si128 (__m128i s1, __m128i s2)

Returns 1 if the following conditions are true: bitwise operation of s1
AND s2 does not equal all zeros and bitwise operation of NOT s1 AND s2
does not equal all zeros, otherwise returns 0. That is,

_mm_testnzc_si128 := ( ( (s1 & s2) != 0 && (~s1 & s2) != 0 ) ? 1 : 0 )

This intrinsic checks if both the CF and ZF flags are not 1 as a result
of the instruction PTEST s1, s2. For example, it allows you to check that
the result has both zeros and ones in s1 on positions specified as set
bits in s2 (mask).

Corresponding instruction: PTEST

Don't really need a multiply.
Build a table of 36 (or 37) 8-bit adds iteratively.
Each thread takes every nth number.
37 threads is good. More than 37 and every so often you'd do two DWORD
adds on the same iteration.
Thread i would do:

    volatile uint64_t incumbent = ~0;
    uint64_t cycles;

    //
    // Given Source a, BYTE adder b, and DWORD adder d
    //
    t = 0;
    for (i = 0; i < 37; i++) {
	table[i] = t;
	t += b;
    }

    initial = a;
    cycles = tnum;
    if (tnum == 0) {			// Special case to skip cycle 0...
	a += table[36];			// can we avoid this with a shift by 1?
	a += d;
	cycles += 37;
    }
    for (; cycles < incumbent; cycles += 37) {
	a += table[tnum];		// Offset to our bucket
	if (a == 0 || a == initial) {	// Exit criteria
	    lock mutex;
	    if (cycles < incumbent)
		incumbent = cycles;
	    unlock mutex;
	    break;
	}
	a += table[36-tnum];		// From our bucket to end
	a += d;				// Special DWORD addition
    }

register __m128i gStart, gByteStep, gDwordStep;

static void *threadmain(void *d)
{
    int tnum = (pthread_t *)d - gThreads;
    static volatile uint64_t incumbent = ~0;
    register uint64_t cycles = tnum;

    register __m128i a, front, back, allones;
    a = _mm_xor_epi8(a, a);
    for (int i = 0; i < 37; ++i) {
	if (i == tnum) front = a;
	else if (i == 36 - tnum) back = a;
	a = _mm_add_epi8(a, gByteStep);
    }
    allones = _mm_cmpeq_epi64(allones, allones);

    if (tnum == 0) {
	a = _mm_add_epi8(a, back);
	a = _mm_add_epi32(a, gDwordStep);
	cycles += 37;
    }

    do {
	a = _mm_add_epi8(a, front);	// Offset to our bucket
	if (_mm_testz_si128(a, a) ||
	    _mm_testc_si128(_mm_cmpeq_epi32(a, gStart), allones)) goto found;
	a = _mm_add_epi8(a, back);	// From our bucket to the end
	a = _mm_add_epi32(a, gDwordStep);
	cycles += 37;
    } while (cycles < incumbent);

    pthread_exit(NULL);			// Another thread found a solution

found:
    pthread_mutex_lock(&gMutex);
    if (cycles < incumbent)
	incumbent = cycles;
    pthread_mutex_unlock(&gMutex);
    pthread_exit(NULL);
}
